132     private BufferedImage lataa(String tiedosto) {
133         BufferedImage palautus = null;
134         try {                
135             palautus = ImageIO.read(new File(tiedosto));
136         } catch (IOException ex) {
137             // handle exception...
138         }
139         return palautus;
140     }
