3   /**
4    * Luokka muuttaa piirrettävän taulukon "lennossa" sellaiseen ulkoasuun, jossa viereiset näkymättömät
5    * ruudut eivät paljasta, mitä viereisessä ruudussa on.
6    */
7   public class PiirrettavanMuokkaaja {
8   
9       private int valitseOikeaMaapala(int y, int x, int[][] alkuTaulukko, boolean[][] visible, int satunnaisuus, int tilanne) {
10          if (tilanne >= 20) {
11              return alkuTaulukko[y][x];
12          }
13          boolean vasemmallaVetta = false;
14          boolean alhaallaVetta = false;
15          boolean ylhaallaVetta = false;
16          boolean oikeallaVetta = false;
17          boolean alkuperainen = true;
18          if (y > 0) {
19              if (alkuTaulukko[y - 1][x] < 150 && alkuTaulukko[y - 1][x] != 109 && visible[y - 1][x]) {
20                  ylhaallaVetta = true;
21              }
22              if (!visible[y - 1][x]) {
23                  alkuperainen = false;
24              }
25          }
26          if (y < 19) {
27              if (alkuTaulukko[y + 1][x] < 150 && alkuTaulukko[y + 1][x] != 109 && visible[y + 1][x]) {
28                  alhaallaVetta = true;
29              }
30              if (!visible[y + 1][x]) {
31                  alkuperainen = false;
32              }
33          }
34          if (x > 0) {
35              if (alkuTaulukko[y][x - 1] < 150 && alkuTaulukko[y][x - 1] != 109 && visible[y][x - 1]) {
36                  vasemmallaVetta = true;
37              }
38              if (!visible[y][x - 1]) {
39                  alkuperainen = false;
40              }
41          }
42          if (x < 19) {
43              if (alkuTaulukko[y][x + 1] < 150 && alkuTaulukko[y][x + 1] != 109 && visible[y][x + 1]) {
44                  oikeallaVetta = true;
45              }
46              if (!visible[y][x + 1]) {
47                  alkuperainen = false;
48              }
49          }
50          if (ylhaallaVetta && alhaallaVetta && vasemmallaVetta && oikeallaVetta) {
51              return 190 + satunnaisuus;
52          } else if (ylhaallaVetta && alhaallaVetta && vasemmallaVetta) { // maa oikealla
53              return 182 + satunnaisuus;
54          } else if (ylhaallaVetta && alhaallaVetta && oikeallaVetta) { // maa vasemmalla
55              return 186 + satunnaisuus;
56          } else if (ylhaallaVetta && oikeallaVetta && vasemmallaVetta) { // maa alhaalla
57              return 184 + satunnaisuus;
58          } else if (oikeallaVetta && alhaallaVetta && vasemmallaVetta) { // maa ylhaalla
59              return 180 + satunnaisuus;
60          } else if (oikeallaVetta && vasemmallaVetta) {
61              return 192 + satunnaisuus;
62          } else if (oikeallaVetta && alhaallaVetta) {
63              return 172 + satunnaisuus;
64          } else if (oikeallaVetta && ylhaallaVetta) {
65              return 170 + satunnaisuus;
66          } else if (vasemmallaVetta && ylhaallaVetta) {
67              return 176 + satunnaisuus;
68          } else if (vasemmallaVetta && alhaallaVetta) {
69              return 174 + satunnaisuus;
70          } else if (alhaallaVetta && ylhaallaVetta) {
71              return 194 + satunnaisuus;
72          } else if (alhaallaVetta) {
73              return 164 + satunnaisuus;
74          } else if (ylhaallaVetta) {
75              return 160 + satunnaisuus;
76          } else if (vasemmallaVetta) {
77              return 166 + satunnaisuus;
78          } else if (oikeallaVetta) {
79              return 162 + satunnaisuus;
80          }
81          if (alkuperainen) {
82              return alkuTaulukko[y][x];
83          } else {
84              return 150;
85          }
86      }

88      /**
89       * Muokkaa piirtotaulukosta näytettävän version sen mukaan, paljon maastoa on näkyvillä. Tällä estetään se, että
90       * pelaaja voisi näkyvästä maapalasta päätellä, onko viereisessä ruudussa vettä vai maata.
91       * @param piirrettava   maaston piirtotaulukko.
92       * @param visible       maaston näkyvyystaulukko.
93       * @param satunnaisuus  maaston satunnaisuustaulukko.
94       * @param tilanne       tilanne on erilainen, kun pelin päättyessä kaikki muuttuu näkyväksi. silloin taulukko palautetaan sellaisenaan.
95       * @return              näytettävän piirtotaulukon.
96       */
97      public int[][] piirrettava(int[][] piirrettava, boolean[][] visible, int[][] satunnaisuus, int tilanne) {
98          int[][] palautus = new int[20][20];
99          for (int loop = 0; loop < 400; loop++) {
100             if (visible[loop / 20][loop % 20]) {
101                 if (piirrettava[loop / 20][loop % 20] >= 150) {
102                     palautus[loop / 20][loop % 20] = this.valitseOikeaMaapala(loop / 20, loop % 20, piirrettava, visible, satunnaisuus[loop / 20][loop % 20], tilanne);
103                 } else {
104                     palautus[loop / 20][loop % 20] = piirrettava[loop / 20][loop % 20];
105                 }
106             } else {
107                 palautus[loop / 20][loop % 20] = 8;
108             }
109         }
110         return palautus;
111     }
