229     private BufferedImage lataa(String tiedosto) {
230         BufferedImage palautus = null;
231         try {
232             InputStream is = getClass().getClassLoader().getResourceAsStream(tiedosto);
233             palautus = ImageIO.read(is);
234         } catch (IOException ex) {
235             // handle exception...
236         }
237         return palautus;
238     }
