26      @Override
27      public void run() {
28          frame = new JFrame("Battle of the Baltic Sea 2020");
29          frame.setPreferredSize(new Dimension(1220, 700));
30          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
31  
32          Random arpoja = new Random();
33          
34          pelirunko = new PeliRunko(arpoja);
35          nt = new NappainTapahtuma();
36          nt.setFrame(frame);
37          luoKomponentit(frame.getContentPane());
38          lisaaKuuntelijat();
39          
40          frame.pack();
41          frame.setVisible(true);
42      }
43  
44      private void luoKomponentit(Container container) {
45          piirtoalusta = new Grafiikka(pelirunko);
46          container.add(piirtoalusta);
47          this.grafik = piirtoalusta;
48      }
49  
50      private void lisaaKuuntelijat() {
51          frame.addKeyListener(new NappaimistonKuuntelija(this.grafik, pelirunko, nt));
52      }
53  
54      public JFrame getFrame() {
